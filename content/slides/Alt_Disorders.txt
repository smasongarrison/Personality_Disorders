\frametitle{Dimensional Personality Disorders}
\begin{columns}
\begin{column}{6cm}
\begin{itemize}
\item<1->{Antisocial personality disorder}
\alt<1>{\begin{itemize}\item typical features include ``a failure to conform to lawful and ethical behavior, and \item an egocentric, callous lack of concern for others, \item accompanied by deceitfulness, irresponsibility, \item manipulativeness, and/or risk taking.''\end{itemize}}{\item<2-> Avoidant personality disorder}
\alt<2>{\begin{itemize}\item features include ``avoidance of social situations and inhibition in interpersonal relationships \item related to feelings of ineptitude and inadequacy, \item anxious preoccupation with negative evaluation and rejection, and \item fears of ridicule or embarrassment.''\end{itemize}}{\item<3-> Borderline personality disorder}
\alt<3>{\begin{itemize}\item features are ``instability of self-image, personal goals, interpersonal relationships, and affects, \item accompanied by impulsivity, risk taking, and/or hostility.''\end{itemize}}{ }
\end{itemize}

\begin{itemize}
\item<1->{Narcissistic personality disorder}
\alt<1>{\begin{itemize}\item typical features include ``variable and vulnerable self-esteem, \item with attempts at regulation through attention and approval seeking, and either 
\item overt or covert grandiosity.''\end{itemize}}{\item<2-> Obsessive-compulsive personality disorder}
\alt<2>{\begin{itemize}\item features include ``difficulties in establishing and sustaining close relationships, \item associated with rigid perfectionism, inflexibility, and \item restricted emotional expression.''\end{itemize}}{ }
\end{itemize}
